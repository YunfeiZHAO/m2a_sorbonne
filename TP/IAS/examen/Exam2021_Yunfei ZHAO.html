# Answer
def loss(X):
    mask = X < -1
    l1 = mask * (-4 * X)
    l2 = ~mask * np.maximum(0, 1 - X) ** 2 
    return l1 + l2

def diff_loss(X):
    mask = X < -1
    l1 = mask * (-4)
    l2 = ~mask * (-2 * (1-X))

# Answer
def obj(w, b, data):
    X, y, alpha = data
    n = X.shape[0]
    r = (alpha/2) * w@w.T + (1/n) * np.sum(loss((y * (X@w + b))))
    return r
